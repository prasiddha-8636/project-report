% MATHEMATICAL AND SCIENTIFIC SYMBOLS DEFINITIONS
% Use \gls{key} in text to reference symbols

\newglossaryentry{symb:theta}{
    type=symbols,
    name={\ensuremath{\theta}},
    description={Azimuth angle in degrees, ranging from $0^\circ$ to $360^\circ$}
}

\newglossaryentry{symb:phi}{
    type=symbols,
    name={\ensuremath{\phi}},
    description={Zenith angle or elevation of the mountain peak}
}

\newglossaryentry{symb:lat}{
    type=symbols,
    name={\ensuremath{\phi_{lat}}},
    description={Geographic latitude of the camera viewpoint}
}

\newglossaryentry{symb:lon}{
    type=symbols,
    name={\ensuremath{\lambda_{lon}}},
    description={Geographic longitude of the camera viewpoint}
}

\newglossaryentry{symb:dist}{
    type=symbols,
    name={\ensuremath{d(Q, R)}},
    description={Distance metric (Euclidean or DTW cost) between query $Q$ and reference $R$}
}

\newglossaryentry{symb:height}{
    type=symbols,
    name={\ensuremath{h(x, y)}},
    description={Terrain elevation at coordinates $(x, y)$ from the DEM}
}
